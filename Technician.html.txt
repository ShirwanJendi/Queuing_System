    <script>
        $(document).ready(function () {
            const socket = io();

            $('#info-form').submit(function (event) {
                event.preventDefault();
                const formData = {
                    number: $('#number').val(),
                    device_name: $('#device_name').val(),
                    repair_info: $('#repair_info').val(),
                    repair_time: $('#repair_time').val(),
                    created_at: new Date().toISOString().slice(0, 19).replace('T', ' ')
                };
                $.ajax({
                    url: '/save_info',
                    type: 'POST',
                    contentType: 'application/json',
                    data: JSON.stringify(formData),
                    success: function (response) {
                        console.log('Info saved successfully');
                        $('#info-form')[0].reset(); // Reset the form after successful submission
                        fetchLatestNumbers(); // Update the displayed numbers
                    }
                });
            });

            function updatePressedNumbers(numbers) {
                $('#pressed-numbers').empty();
                numbers.forEach(item => {
                    $('#pressed-numbers').append(
                        `<div class="pressed-number">
                            <span class="number">${item.number}</span>
                            <span class="timestamp">${item.timestamp}</span>
                            <button class="display-btn" data-number="${item.number}">Display</button>
                        </div>`
                    );
                });
                $('.display-btn').click(function() {
                    const number = $(this).data('number');
                    $.ajax({
                        url: '/display_number',
                        type: 'POST',
                        contentType: 'application/json',
                        data: JSON.stringify({ number: number }),
                        success: function (response) {
                            console.log('Number displayed on TV');
                        }
                    });
                });
            }

            socket.on('number_update', function (msg) {
                fetchLatestNumbers();
            });

            function fetchLatestNumbers() {
                $.ajax({
                    url: '/get_latest_numbers',
                    type: 'GET',
                    success: function (data) {
                        updatePressedNumbers(data);
                    }
                });
            }

            setInterval(fetchLatestNumbers, 5000); // Fetch latest numbers every 5 seconds
        });
    </script>
